    物体检测打标小脚本-获取当前图片中鼠标位置

    编程 编程 python 技术小结
    创建时间:2019-09-05 15:30
    阅读:
    评论:

    用python实现图片中鼠标位置并记录以其为中心的正方形检测框（只有左上，右下坐标点）

    老师给布置了个任务：这有一个文件夹里面有一些图片，使用python将图片显示出来，并且实时记录鼠标的位置，当鼠标点击图片的时候，将该位置记录下来并存于txt文本中。(图片在py文件的同级文件夹中)

    本文改自CSDN上面Big_Head_ 博主的博客

    原链接：https://blog.csdn.net/Big_Head_/article/details/78966363

    以下为代码：

    import numpy as np
import matplotlib.animation as animation
import matplotlib.pyplot as plt
from os import walk
from PIL import Image
import os
import signal
def get_lurd(point_dir,pic_dir):
    f = open(point_dir, "a")
    #f.write(filename)
    f.close()
    for (dirpath, dirnames, filenames) in walk(pic_dir):
        f = []
        f.extend(filenames)
        print(f)
        print(len(f))
    for i in range(0,len(f)):
        filename=pic_dir+f[i]
        print(filename)

        def on_press(event):#event为事件，在这里是点击鼠标这个事件
            f=open(point_dir,"a")
            print("当前鼠标位置:" ,event.button,event.xdata, event.ydata)
            lu="左上 :("+str(event.xdata-12)+','+str(event.ydata-12)+')'
            rd="右下 :("+str(event.xdata+12)+','+str(event.ydata+12)+')'
            print(lu)
            print(rd)
            f.write(filename)
            f.write('    ')
            f.write(lu)
            f.write('    ')
            f.write(rd)
            f.write('\r\n')

        fig = plt.figure()#用matplotlib给出一个plot对象
        img = Image.open(filename)#定义这个图像是0.png,  updata = True
        plt.imshow(img, animated= True)#animated这个参数在网上没有找到具体的解释，但是应该是跟动有关的设置参数
        fig.canvas.mpl_connect('button_press_event', on_press)#button_press_event检测到鼠标点击的位置的这个动作
        plt.show()#将图片显示出来

if __name__ == "__main__":
    basedir = os.path.abspath(os.path.dirname(__file__))#获取当前的绝对路径
    print(basedir)
    pointinfo_dir=basedir+'\point_info.txt'
    pic_dir = basedir+'//pic//'
    get_lurd(pointinfo_dir,pic_dir)#指定图片文件夹位置和点信息的位置

    import pygame
import matplotlib.pyplot as plt
from pygame.locals import *
from os import walk
from sys import exit
from PIL import Image
import os
#此小程序的目的是为了刷新出图片，用鼠标点击觉着合适的目标进行目标检测，将鼠标所在周围24*24的框的点信息记录下来
#任意键盘输入都是下一张图片，鼠标左键记录点信息
#遍历完所有图片后在脚本的对话框会出现“图片处理完毕”字样，或者感觉键盘按不动了也是处理完了
#在记录信息时，最好是一次性处理完~~
__author__ = {'name' : 'Hongten',
       'mail' : 'hongtenzone@foxmail.com',
       'QQ'  : '648719819',
       'Version' : '1.0'}#原作者的联系方式
f = []#f为一个列表，目的是存放文件夹里图片文件的名字，放在外面的意思是可以共用
def get_lurd(point_dir,pic_dir):#传入 需要记录点信息的txt文件和图片文件夹
    # print(point_dir)
    # print(pic_dir)测试用看是否能输出文件夹
    for (dirpath, dirnames, filenames) in walk(pic_dir):#先将文件夹遍历一遍将filenames都放入f这个列表
        f.extend(filenames)
    for i in range(0, len(f)):
        BG_IMAGE=pic_dir+f[i]#先将背景定义出来
        basedir = os.path.abspath(os.path.dirname(__file__))  # 获取当前的绝对路径
        MOUSE_IMAGE = basedir+'\mouse.png'#这里的处理方法是用美图秀秀把一个空白的图片扣掉中间的矩形，使其成为中间透明的方框，再在底下替换掉鼠标
        pygame.init()
        # 设置窗口的大小
        screen = pygame.display.set_mode((224, 224), 0, 32)#图片大小为224*224
        pygame.display.set_caption('框的位置')#出来界面的名字
        bg = pygame.image.load(BG_IMAGE).convert()#背景设置
        mouse_cursor = pygame.image.load(MOUSE_IMAGE).convert_alpha()#鼠标设置
        while True:
            for event in pygame.event.get():#一直监控这pygame里的event
                if event.type == QUIT:#退出
                    exit()
                if event.type == pygame.KEYDOWN:
                    i+=1
                    try:
                        BG_IMAGE=pic_dir+f[i]#这里的try是为了防止数组越界
                    except IndexError as e:
                        print(e)
                        print('警告：图片处理完毕')
                    print("下一张图片")
                    bg = pygame.image.load(BG_IMAGE).convert()
                if event.type == MOUSEBUTTONDOWN:
                    x,y=pygame.mouse.get_pos()
                    lu="左上 :("+str(x-12)+','+str(y-12)+')'
                    rd = "右下 :(" + str(x + 12) + ',' + str(y + 12) + ')'
                    pointfile = open(point_dir, "a")#打开文件准备记录点的位置
                    try:#这里的try是为了防止数组越界
                        print(f[i])
                    except IndexError as e:
                        print(e)
                        print('警告：图片处理完毕')
                    try:#这里的try是为了防止数组越界
                        pointfile.write(f[i])
                        pointfile.write('    ')
                        pointfile.write(lu)  # left&up
                        pointfile.write('    ')
                        pointfile.write(rd)  # right&down
                        pointfile.write('\r\n')
                        print(lu)
                        print(rd)
                    except IndexError as e:
                        print(e)
                        print('警告：图片处理完毕')

            screen.blit(bg, (0, 0))
            # 鼠标的x,y坐标
            x, y = pygame.mouse.get_pos()
            # 隐藏鼠标
            pygame.mouse.set_visible(False)
            x -= mouse_cursor.get_width() / 2#这两行是根据mouse_cursor的宽和高调整x和y数值
            y -= mouse_cursor.get_height() / 2
            # 用其他图形代替鼠标
            screen.blit(mouse_cursor, (x, y))
            pygame.display.update()
if __name__ == "__main__":
    basedir = os.path.abspath(os.path.dirname(__file__))#获取当前的绝对路径
    pointinfo_dir=basedir+'\point_info.txt'#这几行是对文件夹的位置进行字符串的操作
    pic_dir = basedir+'//pic//'
    get_lurd(pointinfo_dir,pic_dir)#指定图片文件夹位置和点信息的位置
