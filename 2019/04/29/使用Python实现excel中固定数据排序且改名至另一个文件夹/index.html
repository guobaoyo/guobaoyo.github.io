    用python实现excel中读取文件目录+相应数据排序+将文件重命名（学姐写的）

    20190429老师让我把最后得到的result.csv文件中波点，豹纹，条纹的前100张图片分别弄到三个文件夹里面，想看看效果。

    我python贼烂，下面的这个代码是学姐（半个小时之内）写的，我用一下午学会了，在这里感谢学姐并做个记录以后行了再自己慢慢写。

    文件结构如下：

    • 先将第一行和第一列删掉再处理
    • 最后一个for循环是为了将图片的名字改为概率大小，这样看起来比较方便
    import os
from PIL import Image
import shutil
import pandas as pd
path_csv="./tiaowen.csv"#尽量用英文名字，这样会可靠一些
path=""
df = pd.read_csv(path_csv, header=None)#先将原来的文件读取并且称其为df
df.columns = ['filename', 'tiaowen']#实际上将表格的两个列定义为filename 和 tiaowen
list1=sorted(df.tiaowen)#从小到大排序,而使用sort()方法对list排序会修改list本身，不会返回新的list，而sorted()会返回一个新的list，并不会改变原来的list
#print('**************')
for i in (sorted(df.tiaowen)[-100:]):#看看倒数100个都是多少评分
    print(i)
#print('**************')
df1=df[df.tiaowen>0.999528]#根据刚才的倒数第100个的分数决定数值
#print(df1.filename)
list2=[]#新建个list2这个列表
for i in df1.tiaowen:#
    list2.append(i)#将tiaowen这个  概率  加入到list2这个列表中
print(list2)#
print('**************')

newpath="./上衣/tiaowen1"#新建一个tiaowen的文件夹
for i in df1.filename:
    print(i)
    try:
        shutil.copy(i,newpath)#i为源路径，newpath为目标路径，在这里做一个try和except目的是将报错的文件跳过去（之前不用try的时候会发生文件重名的bug）
    except FileNotFoundError as e:
        continue
j=0#下面的这个for循环的作用是将图片的名字改为概率.jpg
for file_name in os.listdir(newpath):#在这里就将file_name全变为newpath里面的文件名字

    os.rename(os.path.join(newpath,file_name),os.path.join(newpath,str(list2[j])+".jpg"))
    j+=1#上面这句话最好加个try和except的语句这样就可以避免报错

    思路二：既然是要首字母为y那我在excel中将第三列升序排列，只要后面的首字母为y的就ok了。故如法炮制，在前一天的小工具上面改。

    import os
from PIL import Image
import shutil
import pandas as pd
path_csv="./shouy-train2.csv"#尽量用英文名字，这样会可靠一些
path="./fashionAI_attributes_train2//"
df = pd.read_csv(path_csv, header=None)#先将原来的文件读取并且称其为df
df.columns = ['filename', 'leibie','tag']#实际上将表格的两个列定义为filename 和 tiaowen
list1=sorted(df.leibie)#从小到大排序,而使用sort()方法对list排序会修改list本身，不会返回新的list，而sorted()会返回一个新的list，并不会改变原来的list
newpath="./train2/"#新建一个tiaowen的文件夹
zimulu = str(df.leibie)
print(newpath+zimulu)
list2=[]#新建个list2这个列表
for z in df.leibie:#
    list2.append(z)#将tiaowen这个名字加入到list2这个列表中
print(list2[4])
j=0#设置j=0
for i in df.filename:
    try:
        shutil.copy(os.path.join(path+i),os.path.join(newpath+list2[j]))
        j += 1#每次都让j加1，这样就可以对应上list[]这个列表了
    except FileNotFoundError as e:
        continue
