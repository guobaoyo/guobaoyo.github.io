    ​ 2.2下载文件（python模拟http-get请求下载文件）：

    #下面的是下载文件，需要输入想要下载的文件名
import requests
import os
def downloadfile(file_name):
    real_url = 'http://localhost:8080/download/'+file_name
    r = requests.get(real_url, stream=True)
    print(r)
    basedir = os.path.abspath(os.path.dirname(__file__))  # 获取当前文件的绝对路径
    save_dir = basedir + '\\' + 'download'
    if os.path.exists(save_dir) is False:
        os.makedirs(save_dir)
    f = open("./"+'download'+'\\'+file_name, "wb")
    for chunk in r.iter_content(chunk_size=512*1024):
        if chunk:
            f.write(chunk)
    print(file_name+"下载成功")
if __name__ == '__main__':
    downloadfile(file_name='testpic.jpg')

    2.3服务器端（处理上传请求和下载请求写在了一起，用不同的处理器去处理）：

    package main

import (
   "fmt"
   "io"
   "log"
   "net/http"
   "os"
)

func main() {
   http.Handle("/download/",http.StripPrefix("/download/", http.FileServer(http.Dir("src/httprouter-tutorial-master/part4/download"))))//这里写上下载文件的路由。StripPrefix 函数将会过滤掉/download/，并将修改过的请求定向到http.FileServer所返回的Handler中去,cichu
   http.HandleFunc("/upload/",uploadFileHandler)//这里写上上传文件的路由条目
   log.Fatal(http.ListenAndServe(":8080", nil))//监听8080端口
}

func uploadFileHandler(w http.ResponseWriter, r *http.Request){
   file, handler, err := r.FormFile("file")
   if err != nil {
      fmt.Println("问题出在解析文件")//可能是解析不出来，在Python文件中如果后面没有/，就会出现body为none的情况
      fmt.Println(err)
      fmt.Println("问题出在解析文件")
      return
   }
   defer file.Close()//延迟这个文件关闭
   fmt.Fprintf(w, "%v", handler.Header)
   f, err := os.OpenFile("C:\\Go\\mycode\\src\\httprouter-tutorial-master\\part4\\upload\\"+handler.Filename, os.O_WRONLY|os.O_CREATE, 0666) //1.这里是绝对路径，记得到服务器端配置时要更改路径。此处假设当前目录下已存在upload目录
   if err != nil {//对Os.openfile进行判断，看是否会有错
      println("问题出在打开文件拷贝到本地")
      fmt.Println(err)//打印出来这个错误
      println("问题出在打开文件拷贝到本地")
      return
   }
   defer f.Close()//延迟一会儿f的关闭
   io.Copy(f, file)//把延迟的那两个文件进行拷贝到本地
   println("上传文件"+handler.Filename+"成功")
}

    3.4学习怎么用gon列出来指定文件夹下面文件的MD5值，并且把文件名和MD5值都存入字典里面：

    //下面是列出文件方法
package main

import (
   "crypto/md5"
   "encoding/hex"
   "fmt"
   "io/ioutil"
)

func main() {
   folder := "./src/httprouter-tutorial-master/part4/download/"//设置想要查看的文件夹
   listFile(folder)
}

func MD5Bytes(s []byte) string {
   ret := md5.Sum(s)
   return hex.EncodeToString(ret[:])
}

//计算文件MD5值
func MD5File(file string) (string, error) {
   data, err := ioutil.ReadFile(file)
   if err != nil {
      return "", err
   }
   return MD5Bytes(data), nil
}

func listFile(folder string){
   m := make(map[string]string)
   files, _ := ioutil.ReadDir(folder) //specify the current dir
   for _,file := range files{
      if file.IsDir(){
      }else{
         testFile1 := "./src/httprouter-tutorial-master/part4/download/"+file.Name()
         md5string, _ := MD5File(testFile1)
         //println("这是文件"+testFile1+md5string)
         m[file.Name()] = md5string
      }
   }
   fmt.Println(m)
}

        server_name_MD5_dict = eval(server_name_md5_temp_string)  # 将字符串的形式转为字典形式

    while True:
        b = f.read(8096)
        if not b :
            break
        myhash.update(b)
    f.close()
    return myhash.hexdigest()
