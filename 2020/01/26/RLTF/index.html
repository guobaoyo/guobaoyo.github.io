    1.第二章硬币问题的概率与熵的关系图

    import numpy as np
import mathimport matplotlib.pyplot as plt
def coin_entropy():
    x = np.linspace(0,1,101)
    print(x)
    y = -x*np.log2(x)-(1-x)*np.log2(1-x)
    print(y)
    y[np.isnan(y)] = 0
    print(y)
    plt.plot(x,y)
    plt.show()
def main():
    coin_entropy()
if __name__ == '__main__':
    main()

    y[np.isnan(y)] = 0的作用是使得y中所有的nan都转化为0

    2.Gaussian_KL

    import numpy as np
from mpl_toolkits.mplot3d import Axes3D
import matplotlib.pyplot as plt
def Gaussian_KL():
    fig = plt.figure()
    ax = Axes3D(fig)
    x = np.linspace(0.1,2,31)
    y = np.linspace(-2,2,31)
    X,Y = np.meshgrid(x,y)
    Z = -np.log(X)+X*X+Y*Y/2-0.5
    ax.plot_surface(X,Y,Z,rstride=1,cstride=1,cmap='rainbow')
    plt.show()
def main():
    Gaussian_KL()
if __name__ == '__main__':
    main()

    其中X,Y = np.meshgrid(x,y)是针对x,y生成网格点坐标矩阵,将x,y的31对儿点连接成坐标矩阵。

    rstride指的是行步长，cstride指的是列步长，步长越大图像越粗糙，且步长不能为小数。

    参考:https://blog.csdn.net/ygdxt/article/details/86618606

    3.[存疑，未解决 ]momentum与nesterov对比：

    def momentum(x_start,step,g,discount = 0.7):
    x = np.array(x_start,dtype='float64')
    print(x)
    pre_grad = np.zeros_like(x)
    for i in range(50):
        grad = g(x)
        pre_grad = pre_grad*discount + grad * step
        x -= pre_grad
        print('[epoch {0}] grad = {1} , x = {2} '.format(i,grad,x))
        if abs(sum(grad))<1e-6:
            break
    return x

    def nesterov(x_start,step,g,discount = 0.7):
    x = np.array(x_start,dtype='float64')
    pre_grad = np.zeros_like(x)
    for i in range(50):
        x_future = x - step*discount*pre_grad
        grad = g(x_future)
        pre_grad = pre_grad*0.7 + grad
        x -= pre_grad * step
        print('[epoch {0}] grad = {1} , x = {2} '.format(i,grad,x))
        if abs(sum(grad)) < 1e-6:
            break

    ①存在的问题是我只是画出来了U型的弯道，但没有像书中一样画出上面递进更新的点（这是其次）

    ②重要的是对比上述两算法 ，nesterov为什么👇？

    pre_grad = pre_grad*0.7 + grad
x -= pre_grad * step

    按照momentum的理解方式，不应该是👇？

    pre_grad = pre_grad*0.7 + grad*step
x -= pre_grad

    解决：将momentum算法做对比实验，分别将step参数如上面两种方式进行计算，所得结果一样：

    但是这是为什么呢？不合常人理解啊？

    x = np.array(x_start,dtype='float64')
pre_grad = np.zeros_like(x)

    目的是为了照顾底下数组形式的输入.
