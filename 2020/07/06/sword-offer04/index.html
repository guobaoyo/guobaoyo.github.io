    class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        if len(matrix)==0 or len(matrix[0])==0:
            return False
        m = len(matrix)
        n = len(matrix[0])
        left = 0
        right = m*n - 1
        mid = (left+right)//2
        while left<right:
            mid = (left+right)//2
            # print('*')
            # print(left)
            # print(right)
            # print(mid)
            # print('*')
            if matrix[mid//n][mid%n]>target:
                # print('å¤§äº†')
                right = mid
            elif matrix[mid//n][mid%n]<target:
                # print('å°äº†')
                left = mid + 1
            else:
                left = mid
                break
        #print(left)
        if matrix[left//n][left%n]==target:
            return True
        return False
